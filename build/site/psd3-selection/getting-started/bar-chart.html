Building a Bar Chart

This tutorial is coming soon. It will cover:

Preview

barChart :: Array Number -> A.AST Number
barChart data' =
  A.elem SVG
    [ width (num 500.0)
    , height (num 300.0)
    ]
    `A.withChild`
      A.joinData "bars" "rect" data' \d ->
        A.elem Rect
          [ from "x" (\_ i -> toNumber i * 25.0)
          , from "y" (\val _ -> 300.0 - val)
          , width (num 20.0)
          , from "height" (\val _ -> val)
          , fill (text "steelblue")
          ]
